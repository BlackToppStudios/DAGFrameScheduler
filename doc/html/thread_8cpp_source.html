00123         // Thread wrapper function.
00124         #if defined(_MEZZ_THREAD_WIN32_)
00125         unsigned WINAPI thread::wrapper_function(void * aArg)
00126         #elif defined(_MEZZ_THREAD_POSIX_)
00127         void * thread::wrapper_function(void * aArg)
00128         #endif
00129         {
00130           // Get thread startup information
00131           _thread_start_info * ti = (_thread_start_info *) aArg;
00132 
00133           try
00134           {
00135             // Call the actual client thread function
00136             ti->mFunction(ti->mArg);

00140             // Uncaught exceptions will terminate the application (default behavior

00146           lock_guard<mutex> guard(ti->mThread->mDataMutex);

00149           // The thread is responsible for freeing the startup information
00150           delete ti;
