202  class MEZZ_LIB Thread::id

221  /// @brief Assignment Operator.
222  /// @param aId The right hand operand during assignment.
223  /// @return A reference to the left hand operand after it has been assigned.
224  inline id & operator=(const id &aId)
225  {
226  mId = aId.mId;
227  return *this;
228  }

230  /// @brief Equality Comparison.
231  /// @param aId1 Left hand operand.
232  /// @param aId2 Right hand operand.
233  /// @returns True boolean value if the OS specific IDs match and false otherwise.
234  inline friend bool operator==(const id &aId1, const id &aId2)
235  { return (aId1.mId == aId2.mId); }
236 
237  /// @brief Inequality Comparison.
238  /// @param aId1 Left hand operand.
239  /// @param aId2 Right hand operand.
240  /// @returns False boolean value if the OS specific IDs match and true otherwise.

242  { return (aId1.mId != aId2.mId); }
243 
244  /// @brief Greater than or equal to Comparison.

247  /// @returns A consistent value to allow for sorting/ordering of threads.
