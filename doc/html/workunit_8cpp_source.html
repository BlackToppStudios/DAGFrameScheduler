00149         RunningState WorkUnit::TakeOwnerShip()
00150         {
00151             // We check that all the dependencies are met before trying to search the framescheduler
00152             for (std::vector<WorkUnit*>::iterator Iter = Dependencies.begin(); Iter!=Dependencies.end(); ++Iter)
00153             {
00154                 if( Complete != (*Iter)->GetRunningState() )
00155                     { return NotStarted; }
00156             }
00157 
00158             if(NotStarted ==  AtomicCompareAndSwap(&CurrentRunningState, NotStarted, Running) )
00159                 { return Starting;} // This is the only place a starting should be generated, and it is never placed in CurrentRunningState

00161             return NotStarted;
