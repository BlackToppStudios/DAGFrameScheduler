00085     template <typename RecordType> class BufferedRollingAverage : public RollingAverage<RecordType>
00086     {
00087         protected:

00093             typename std::vector<RecordType>::iterator Current;

00095             /// @brief Move Current to the next place to be written

00097             {
00098                 if(Records.end() == ++Current)
00099                     { Current = Records.begin(); }
