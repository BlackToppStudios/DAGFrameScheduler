100  /// @brief A RollingAverage that stores a copy of each record. and does the math when queried
101  template <typename RecordType> class BufferedRollingAverage : public RollingAverage<RecordType>
102  {
103  protected:
104  /// @brief The collection of all the records that are being have been added going back as far as the capacity will allow.
105  /// @note All writes to this are performed through the iterator @ref Current
106  std::vector<RecordType> Records;
107 
108  /// @brief The iterator used to treat an std::vector as a circular buffer.
109  typename std::vector<RecordType>::iterator Current;
110 
111  /// @brief Move Current to the next place to be written
112  virtual void IncrementIterator()
113  {
114  if(Records.end() == ++Current)
115  { Current = Records.begin(); }
116  }
117 
118  public:
119  /// @brief Used for accessing the derived type when it may not be directly known.
120  typedef RecordType Type;
121 
122  /// @brief Constructor
123  /// @param RecordCount The capacity of the Rolling average. This defaults to 10.
124  BufferedRollingAverage(const Whole& RecordCount = MEZZ_FRAMESTOTRACK)
125  {
126  Records.assign(RecordCount,RecordType(0));
127  Current = Records.begin();
128  }
129 
130  /// @brief Gets the capacity of this rolling average. How many records will this store.
131  virtual Whole RecordCapacity() const
132  { return Records.size(); }
133 
134  /// @brief Inserts a new record into the rolling average and if needed drops the oldest one out.
135  /// @param Datum The record to add.
136  virtual void Insert(RecordType Datum)
137  {
138  IncrementIterator();
139  *Current = Datum;
140  }

142  /// @brief Add up all the records and divide by the count. A simple arithmetic mean.
143  /// @return This returns the current average of all the records as a RecordType. If it is an Integer type floating parts are truncated.
144  virtual RecordType GetAverage() const
145  {
146  RecordType Results(0);
147  for(typename std::vector<RecordType>::const_iterator Iter = Records.begin(); Iter!=Records.end(); ++Iter)
148  {Results += *Iter;}
149  return Results/Records.size();
150  }

205  /// @brief A weighted average that does math with every insertion and stores nothing.
206  /// @details This is possibly an optimization over the BufferedRollingAverage. It has not yet been shown whether the amount of memory for
207  /// each task will significantly affect load times (From RAM to CPU cache) when starting a rolling average, or whether any
208  /// innaccuracies in sorting using this would outweigh that.
209  template <typename RecordType, typename MathType> class WeightedRollingAverage : public RollingAverage<RecordType>

238  /// @brief Update the currently stored Rolling average with a new data point/record.
239  /// @param Datum Update the Current Average according to the following formula CurrentAverage = CurrentAverage * ((RecordCount-1)/RecordCount) + Datum/RecordCount.
240  virtual void Insert(RecordType Datum)
241  {
242  LastEntry = Datum;
243  CurrentAverage = ( MathType(this->CurrentAverage) ? MathType(this->CurrentAverage) : MathType(1)) // A zero really screws with averages that tend to move away from zero
244  * ((MathType(WeightCount)-MathType(1))/MathType(WeightCount)) // Get weight of all the older members
245  + MathType(Datum)/MathType(WeightCount); // wGet the Weight of the Current Member
246  }
