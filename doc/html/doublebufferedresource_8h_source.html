00067         /// @brief A thread specific resource that uses double buffering to avoid multithreaded synchronization mechanisms.
00068         /// @details It is intended for a @ref Mezzanine::Threading::iWorkUnit "iWorkUnit" like the
00069         /// @ref Mezzanine::Threading::LogBufferSwapper "LogBufferSwapper" that just swaps buffers. This work unit should be
00070         /// configured to depend on every other work unit (that is not also not a buffer swapper, so that it will not conflict
00071         /// with any other double buffered resources).
00072         template<typename T>
00073         class DoubleBufferedResource
00074         {
00075             private:
00076                 /// @brief One of the resources to be buffered
00077                 T* ResourceA;
00078 
00079                 /// @brief The other of the resources to be buffered
00080                 T* ResourceB;
00081 
00082             public:
00083                 /// @brief A Constructor that creates default versions of the resources.
00084                 /// @details This calls the default constructors for type T.
00085                 DoubleBufferedResource() : ResourceA(0), ResourceB(0)
00086                 {
00087                     try
00088                     {
00089                         ResourceA = new T;
00090                         ResourceB = new T;
00091                     }catch(std::exception& e){
00092                         if (ResourceA)
00093                             { delete ResourceA; ResourceA=0; }
00094                         if (ResourceB)
00095                             { delete ResourceB; ResourceB=0; }
00096                         throw e;
00097                     }
00098 
00099                 }

00101                 /// @brief A constructor that takes ownership of the the resources passed.
00102                 /// @param Current This will be used as the first resource for threads to use.
00103                 /// @param Buffer This will wait for the other thread
00104                 DoubleBufferedResource(T* Current, T* Buffer) : ResourceA(Current), ResourceB(Buffer)
00105                     {}
